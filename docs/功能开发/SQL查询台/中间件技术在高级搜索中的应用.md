```java
     * 启动查询审批流程（大数据量查询需要审批）
```

### 1. Redis 缓存中间件

#### 应用场景
- 字段元数据缓存
- 查询结果缓存
- 用户搜索历史

#### 技术实现
```java
@Service
public class SearchCacheService {
    
    @Resource
    private StringRedisTemplate redisTemplate;
    
    private static final String FIELD_METADATA_KEY = "search:fields:{}:{}";
    private static final String QUERY_RESULT_KEY = "search:result:{}";
    private static final int CACHE_EXPIRE_HOURS = 2;
    
    /**
     * 缓存字段元数据
     */
    public void cacheFieldMetadata(String module, String projectId, List<FieldMetadata> fields) {
        String key = StrUtil.format(FIELD_METADATA_KEY, module, projectId);
        redisTemplate.opsForValue().set(key, JSON.toJSONString(fields), 
            CACHE_EXPIRE_HOURS, TimeUnit.HOURS);
    }
    
    /**
     * 缓存查询结果（大数据量查询）
     */
    public void cacheQueryResult(String queryHash, AdvancedSearchResult result) {
        String key = StrUtil.format(QUERY_RESULT_KEY, queryHash);
        redisTemplate.opsForValue().set(key, JSON.toJSONString(result), 
            30, TimeUnit.MINUTES); // 查询结果缓存30分钟
    }
    
    /**
     * 获取缓存的查询结果
     */
    public AdvancedSearchResult getCachedResult(String queryHash) {
        String key = StrUtil.format(QUERY_RESULT_KEY, queryHash);
        String cached = redisTemplate.opsForValue().get(key);
        return StringUtils.isNotBlank(cached) ? 
            JSON.parseObject(cached, AdvancedSearchResult.class) : null;
    }
}
```

#### 缓存策略设计
```java
/**
 * 查询缓存策略
 */
@Component
public class QueryCacheStrategy {
    
    /**
     * 判断是否需要缓存查询结果
     */
    public boolean shouldCache(AdvancedSearchRequest request, long resultCount) {
        // 大数据量查询结果缓存
        if (resultCount > 1000) return true;
        
        // 复杂条件查询缓存
        if (request.getCombine().size() > 5) return true;
        
        // 跨项目查询缓存
        if (request.getProjectIds().size() > 3) return true;
        
        return false;
    }
    
    /**
     * 生成查询缓存键
     */
    public String generateCacheKey(AdvancedSearchRequest request) {
        return DigestUtils.md5Hex(JSON.toJSONString(request));
    }
}
```

```java
@Service
public class SearchEventService {
    
    @Resource
    private KafkaTemplate<String, String> kafkaTemplate;
    
    private static final String SEARCH_AUDIT_TOPIC = "search-audit";
    private static final String LARGE_QUERY_TOPIC = "large-query-process";
    
    /**
     * 发送查询审计事件
     */
    public void sendSearchAuditEvent(AdvancedSearchRequest request, long resultCount, long duration) {
        SearchAuditEvent event = SearchAuditEvent.builder()
            .userId(SessionUtils.getUserId())
            .module(request.getModule())
            .conditions(request.getCombine())
            .resultCount(resultCount)
            .duration(duration)
            .timestamp(System.currentTimeMillis())
            .build();
            
        kafkaTemplate.send(SEARCH_AUDIT_TOPIC, JSON.toJSONString(event));
    }
    
    /**
     * 发送大数据量查询处理事件
     */
    public void sendLargeQueryEvent(String queryId, AdvancedSearchRequest request) {
        LargeQueryEvent event = LargeQueryEvent.builder()
            .queryId(queryId)
            .userId(SessionUtils.getUserId())
            .request(request)
            .status("PROCESSING")
            .build();
            
        kafkaTemplate.send(LARGE_QUERY_TOPIC, JSON.toJSONString(event));
    }
}

/**
 * 大数据量查询异步处理
 */
@Component
public class LargeQueryProcessor {
    
    @KafkaListener(topics = "large-query-process", groupId = "search-processor")
    public void processLargeQuery(ConsumerRecord<String, String> record) {
        LargeQueryEvent event = JSON.parseObject(record.value(), LargeQueryEvent.class);
        
        try {
            // 异步执行大数据量查询
            AdvancedSearchResult result = advancedSearchService.executeQuery(event.getRequest());
            
            // 将结果存储到文件或缓存
            storeQueryResult(event.getQueryId(), result);
            
            // 通知用户查询完成
            notifyQueryComplete(event.getUserId(), event.getQueryId());
            
        } catch (Exception e) {
            log.error("Large query processing failed: {}", event.getQueryId(), e);
            notifyQueryFailed(event.getUserId(), event.getQueryId(), e.getMessage());
        }
    }
}
```

```java
@Service
public class ElasticsearchSearchService {
    
    @Resource
    private ElasticsearchRestTemplate elasticsearchTemplate;
    
    /**
     * 全文搜索
     */
    public List<SearchSuggestion> fullTextSearch(String keyword, String module) {
        NativeSearchQuery searchQuery = NativeSearchQuery.builder()
            .withQuery(QueryBuilders.multiMatchQuery(keyword)
                .field("name", 2.0f)  // 名称字段权重更高
                .field("description", 1.0f)
                .field("tags", 1.5f)
                .type(MultiMatchQueryBuilder.Type.BEST_FIELDS))
            .withFilter(QueryBuilders.termQuery("module", module))
            .withHighlightFields(
                new HighlightBuilder.Field("name"),
                new HighlightBuilder.Field("description")
            )
            .withPageable(PageRequest.of(0, 10))
            .build();
            
        SearchHits<SearchDocument> searchHits = elasticsearchTemplate.search(searchQuery, SearchDocument.class);
        
        return searchHits.stream()
            .map(this::convertToSuggestion)
            .collect(Collectors.toList());
    }
    
    /**
     * 智能搜索建议
     */
    public List<String> getSearchSuggestions(String input, String module) {
        SuggestionBuilder suggestionBuilder = SuggestBuilders
            .completionSuggestion("suggest")
            .prefix(input)
            .size(5);
            
        SearchSourceBuilder searchSourceBuilder = new SearchSourceBuilder()
            .suggest(new SuggestBuilder().addSuggestion("search-suggest", suggestionBuilder))
            .query(QueryBuilders.termQuery("module", module));
            
        // 执行搜索并返回建议
        return executeAndParseSuggestions(searchSourceBuilder);
    }
}
```

#### 技术实现

```java
@Service
public class DynamicQueryBuilder {
    
    /**
     * 构建动态查询条件
     */
    public QueryWrapper<TestCase> buildTestCaseQuery(AdvancedSearchRequest request) {
        QueryWrapper<TestCase> queryWrapper = new QueryWrapper<>();
        
        // 基础条件
        if (CollectionUtils.isNotEmpty(request.getProjectIds())) {
            queryWrapper.in("project_id", request.getProjectIds());
        }
        
        // 动态条件构建
        Map<String, Object> combine = request.getCombine();
        for (Map.Entry<String, Object> entry : combine.entrySet()) {
            String field = entry.getKey();
            Map<String, Object> condition = (Map<String, Object>) entry.getValue();
            String operator = (String) condition.get("operator");
            Object value = condition.get("value");
            
            switch (operator) {
                case "like":
                    queryWrapper.like(field, value);
                    break;
                case "in":
                    queryWrapper.in(field, (Collection<?>) value);
                    break;
                case "between":
                    List<?> values = (List<?>) value;
                    queryWrapper.between(field, values.get(0), values.get(1));
                    break;
                case "gt":
                    queryWrapper.gt(field, value);
                    break;
                case "lt":
                    queryWrapper.lt(field, value);
                    break;
                default:
                    queryWrapper.eq(field, value);
            }
        }
        
        // 排序
        List<OrderItem> orders = request.getOrders();
        if (CollectionUtils.isNotEmpty(orders)) {
            for (OrderItem order : orders) {
                if ("asc".equalsIgnoreCase(order.getType())) {
                    queryWrapper.orderByAsc(order.getName());
                } else {
                    queryWrapper.orderByDesc(order.getName());
                }
            }
        }
        
        return queryWrapper;
    }
}
```

## 性能优化策略

### 1. 查询性能优化

```java
@Service
public class QueryPerformanceOptimizer {
    
    /**
     * 查询性能监控
     */
    @Around("@annotation(QueryMonitor)")
    public Object monitorQuery(ProceedingJoinPoint joinPoint) throws Throwable {
        long startTime = System.currentTimeMillis();
        String queryId = UUID.randomUUID().toString();
        
        try {
            Object result = joinPoint.proceed();
            long duration = System.currentTimeMillis() - startTime;
            
            // 记录查询性能
            recordQueryPerformance(queryId, duration, true);
            
            // 慢查询告警
            if (duration > 5000) { // 超过5秒
                alertSlowQuery(queryId, duration, joinPoint.getArgs());
            }
            
            return result;
        } catch (Exception e) {
            long duration = System.currentTimeMillis() - startTime;
            recordQueryPerformance(queryId, duration, false);
            throw e;
        }
    }
    
    /**
     * 查询结果预估
     */
    public long estimateResultCount(AdvancedSearchRequest request) {
        // 基于历史查询数据和条件复杂度预估结果数量
        // 用于决定是否需要异步处理或缓存
        return statisticsService.estimateCount(request);
    }
}
```

### 2. 缓存策略优化

```java
@Configuration
public class CacheConfiguration {
    
    /**
     * 多级缓存配置
     */
    @Bean
    public CacheManager cacheManager() {
        RedisCacheManager.Builder builder = RedisCacheManager
            .RedisCacheManagerBuilder
            .fromConnectionFactory(redisConnectionFactory())
            .cacheDefaults(cacheConfiguration());
            
        return builder.build();
    }
    
    private RedisCacheConfiguration cacheConfiguration() {
        return RedisCacheConfiguration.defaultCacheConfig()
            .entryTtl(Duration.ofHours(2))  // 2小时过期
            .serializeKeysWith(RedisSerializationContext.SerializationPair
                .fromSerializer(new StringRedisSerializer()))
            .serializeValuesWith(RedisSerializationContext.SerializationPair
                .fromSerializer(new GenericJackson2JsonRedisSerializer()));
    }
}
```

### 1. 查询监控指标

```java
@Component
public class SearchMetricsCollector {
    
    private final MeterRegistry meterRegistry;
    private final Counter queryCounter;
    private final Timer queryTimer;
    private final Gauge activeQueriesGauge;
    
    public SearchMetricsCollector(MeterRegistry meterRegistry) {
        this.meterRegistry = meterRegistry;
        this.queryCounter = Counter.builder("search.query.total")
            .description("Total number of search queries")
            .register(meterRegistry);
        this.queryTimer = Timer.builder("search.query.duration")
            .description("Search query execution time")
            .register(meterRegistry);
        this.activeQueriesGauge = Gauge.builder("search.query.active")
            .description("Number of active queries")
            .register(meterRegistry, this, SearchMetricsCollector::getActiveQueryCount);
    }
    
    public void recordQuery(String module, long duration, boolean success) {
        queryCounter.increment(
            Tags.of(
                "module", module,
                "status", success ? "success" : "error"
            )
        );
        queryTimer.record(duration, TimeUnit.MILLISECONDS);
    }
}
```
