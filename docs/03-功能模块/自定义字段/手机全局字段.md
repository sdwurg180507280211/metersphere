## 📝 完整SQL脚本（已执行）

```sql
-- ============================================
-- 步骤1: 创建全局系统字段
-- ============================================
INSERT INTO custom_field (
    id, 
    name, 
    scene, 
    type, 
    remark, 
    options, 
    `system`, 
    `global`, 
    create_time, 
    update_time,
    create_user,
    project_id,
    third_part
)
SELECT 
    'req-num-field-id-2024-12-01-002',  -- ⭐ 唯一ID
    '手机',                             -- ⭐ 字段名称
    'ISSUE',                             -- ⭐ 场景：ISSUE
    'input',                             -- ⭐ 类型：input（文本框）
    '',                                  -- remark
    '[]',                                -- options
    1,                                   -- ⭐ system=1（系统字段）
    1,                                   -- ⭐ global=1（全局字段）
    unix_timestamp() * 1000,            -- create_time
    unix_timestamp() * 1000,            -- update_time
    NULL,                                -- create_user
    NULL,                                -- ⭐ project_id=NULL（全局字段）
    0                                    -- third_part=0
WHERE NOT EXISTS (
    SELECT 1 FROM custom_field 
    WHERE name = '手机' AND scene = 'ISSUE' AND `system` = 1 AND `global` = 1
);

-- ============================================
-- 步骤2: 关联到Local全局模板
-- ============================================
INSERT INTO custom_field_template (id, field_id, template_id, scene, required, `order`, default_value)
SELECT uuid(), 
       (SELECT id FROM custom_field WHERE name = '手机' AND scene = 'ISSUE' AND `system` = 1 AND global = 1 LIMIT 1),
       '5d7c87d2-f405-4ec1-9a3d-71b514cdfda3',  -- Local模板ID
       'ISSUE', 0, 3, ''  -- order=3，显示在"需求号"后面
WHERE NOT EXISTS (
    SELECT 1 
    FROM custom_field_template cft 
    WHERE cft.template_id = '5d7c87d2-f405-4ec1-9a3d-71b514cdfda3'
      AND cft.field_id = (SELECT id FROM custom_field WHERE name = '手机' AND scene = 'ISSUE' AND `system` = 1 AND global = 1 LIMIT 1)
);

-- ============================================
-- 步骤3: 关联到所有项目级模板（关键！）
-- ============================================
INSERT INTO custom_field_template (id, field_id, template_id, scene, required, `order`, default_value)
SELECT 
    uuid() as id,
    (SELECT id FROM custom_field WHERE name = '手机' AND scene = 'ISSUE' AND `system` = 1 AND global = 1 LIMIT 1) as field_id,
    it.id as template_id,
    'ISSUE' as scene,
    0 as required,
    3 as `order`,  -- order=3，显示在"需求号"后面
    '' as default_value
FROM issue_template it
WHERE it.global = 0  -- ⭐ 只处理项目级别的模板
  AND NOT EXISTS (
      SELECT 1 
      FROM custom_field_template cft 
      WHERE cft.template_id = it.id 
        AND cft.field_id = (SELECT id FROM custom_field WHERE name = '手机' AND scene = 'ISSUE' AND `system` = 1 AND global = 1 LIMIT 1)
  );
```
